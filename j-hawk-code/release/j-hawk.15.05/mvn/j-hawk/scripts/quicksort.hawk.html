function quickSort(var arr[] , var start, var end)
{
    if(end > start)
    { 

        var index = `exec partition(arr,start,end)`   
        var tmp = arr[end]
        arr[end] = arr[index]
        arr[index] = tmp

        exec quickSort(arr,start,index-1)
        exec quickSort(arr,index +1 , end)
    }
}

function partition( var arr[]  , var start , var end)
{
  var pivot = arr[end]
  var forever = 1
  var j = end-1
  var i = start

  while( forever == 1)
  {
      while(arr[j] > pivot)
      {
        j = j-1
        if (j  ==  start)
        {
            break
        }
      }
      while(  arr[i] <= pivot)
      {
        i = i+1
        if( i == end )
         {
            break
         }
      }                
      if ( i < j )
      {
        var tmp = arr[j]
        arr[j] = arr[i]
        arr[i] = tmp
      }
      else
      {
        return i
      }
  }
}
