package org.commons.file;

/**
 *
 * @author manosahu
 */
public class ZipException extends Exception{

    private static final long serialVersionUID = 1L;

    public ZipException() {
    }

    public ZipException(Object msg) {
        super(msg.toString());
    }

    public ZipException(Throwable t) {
        super(t);
    }

    public ZipException(Object msg, Throwable t) {
        super(msg.toString(), t);
    }

}
