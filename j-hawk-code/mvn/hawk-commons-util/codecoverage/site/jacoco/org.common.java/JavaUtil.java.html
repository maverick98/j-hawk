public class JavaUtil {

    public static boolean isNull(Object object) {
        return object == null;
    }

    public static boolean isNull(Object... objects) {
       
        for (Object object : objects) {
            if (!isNull(object)) {
                return false;
            }
        }
        return true;
    }

}
