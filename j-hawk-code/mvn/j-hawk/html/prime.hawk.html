function calculateSquareRoot(var x)
{
return `exec mm->sqrt(x)`
}

//This checks if a number is prime or not
function checkPrime(var x)
{

    for(var i=2; i<`exec calculateSquareRoot(x)`;i=i+1)
    {
        if( x%i == 0)
        {
            return 0
        }
    }

    return 1
}
